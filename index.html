API Documentation for Blog laravel project

Github Repo

signup

post → { /api/register }

name → required

email → required

role → optional → [default = "user"]

is_partner → optional → [default = false]

password → required

password_confirmation → required

login

post → { /api/login }

email → required

password → required

logout

post → { /api/logout }

forgot password

post → { /api/forgot-password }

email → required

reset password

post → { /api/reset-password/{token} }

password → required

password_confirmation → required

current user

get → { /api/user }

All users

get → { /api/allusers }

All employers

get → { /api/employers }

get employer

get → { /api/employers/{id} }

create employer

post → { /api/employers }

user_id → required

edit employer

patch → { /api/employers/{id} }

name → optional

email → optional

role → optional

is_partner → optional

password → optional

password_confirmation → required if there were password

delete employer

delete → { /api/employers/{id} }

All blogs

get → { /api/blogs }

get blog

get → { /api/blogs/{id} }

create blog (multipart form data)

post → { /api/blogs }

category → required

title → required

is_paid → required

section_title → required

content → optional

image → optional

edit blog

patch → { /api/blogs/{id} }

category → optional

title → optional

is_paid → optional

delete blog

delete → { /api/blogs/{id} }

All blog sections

get → { /api/blogs/{blog_id}/sections }

blog section

get → { /api/blogs/{blog_id}/sections/{section_id} }

create blog section (multipart form data)

post → { /api/blogs/{id}/sections }

section_title → required

content → optional

image → optional

edit blog section (multipart form data)

post → { /blogs/{blog_id}/sections/{section_id}/update }

section_title → optional

content → optional

image → optional

delete blog section

delete → { /api/blogs/{blog_id}/sections/{section_id} }
